
Overriding -> implementinf a method in a subclass which is already
implemented in a parent class !

package pack;

public class LivingBeing {

	public String eyes = "twoEyes";
	 String legs = "running";
	
	 public void seeing() {
		System.out.println("Eyes is used to see, ");
	}
	
	 public void run() {
		System.out.println("Legs are used to run, parent class method");
	}
}
